    #!/usr/bin/env bash
    # skills/_shared/_lib.sh
    # Shared color helpers, slug derivation, and atomic file replace.
    # Pure bash 3.2 + POSIX sed/tr. Sourced by init-context.sh and state.sh.
    # No `set -e` here — sourced by callers that own their own pipefail discipline.

    # Color codes (lifted VERBATIM from install.sh:10-18 per D-55)
    GREEN='\033[0;32m'
    YELLOW='\033[1;33m'
    RED='\033[0;31m'
    NC='\033[0m'

    info()  { echo -e "${GREEN}[+]${NC} $1"; }
    warn()  { echo -e "${YELLOW}[!]${NC} $1"; }
    error() { echo -e "${RED}[x]${NC} $1" 1>&2; exit 1; }

    # godmode_slug: free-form title → kebab-case slug. Bash 3.2 portable (no ${var,,}).
    # Algorithm per RESEARCH § "Brief slug derivation (D-22)" and CR-04:
    #   1. tr [:upper:] [:lower:]
    #   2. sed: replace any run of non-[a-z0-9] with single -
    #   3. sed: strip leading/trailing -
    godmode_slug() {
      printf '%s' "${1:-}" \
        | tr '[:upper:]' '[:lower:]' \
        | sed -E 's/[^a-z0-9]+/-/g' \
        | sed -E 's/^-+|-+$//g'
    }

    # godmode_atomic_replace: write content to target via mktemp+mv (POSIX atomic).
    # Args: $1 = content (string); $2 = target path
    # If $1 is the literal string "-" read content from stdin instead.
    godmode_atomic_replace() {
      local content="$1" target="$2" tmp
      tmp=$(mktemp -t godmode-atomic.XXXXXX) || return 1
      if [ "$content" = "-" ]; then
        cat > "$tmp"
      else
        printf '%s' "$content" > "$tmp"
      fi
      mv "$tmp" "$target"
    }
    ```

    Notes:
    - Use `1>&2` for error redirection so info/warn don't pollute stderr (per D-55 UX consistency with installer).

    - `tr '[:upper:]' '[:lower:]'` is POSIX-portable; `${var,,}` is bash 4+ only (CR-04).
    - `sed -E` is BSD/macOS portable for `+`/`?` — same flag works on GNU sed.
    - Do NOT add `set -euo pipefail` here — this file is SOURCED, not executed. Callers (init-context.sh, state.sh) set their own pipefail.

    #   $2 = active_brief_slug (kebab-case, e.g., "skill-layer")
    #   $3 = status (free-form, e.g., "Ready to plan")
    #   $4 = next_command (e.g., "/plan 4")
    #   $5 = audit_line (one-line description appended to audit log)
    godmode_state_update() {
      local active_brief="$1"
      local active_brief_slug="$2"
      local status="$3"
      local next_cmd="$4"
      local audit_line="$5"
      local state_file=".planning/STATE.md"

      [ -f "$state_file" ] || { echo "[x] $state_file not found — run /mission first" 1>&2; return 1; }

      local now_iso
      now_iso=$(date -u +%Y-%m-%dT%H:%M:%SZ)
      local today
      today=$(date -u +%Y-%m-%d)

      # 1. Build new YAML front matter via jq -n --arg (CR-02 discipline; never heredoc)
      local new_fm
      new_fm=$(jq -nr \
        --argjson v 1 \
        --argjson n "$active_brief" \
        --arg slug "$active_brief_slug" \
        --arg s "$status" \
        --arg c "$next_cmd" \
        --arg la "${now_iso} — ${audit_line}" \
        '"---\ngodmode_state_version: \($v)\nactive_brief: \($n)\nactive_brief_slug: \($slug)\nstatus: \($s)\nnext_command: \($c)\nlast_activity: \"\($la)\"\n---"')

      # 2. Preserve audit-log body (everything after the second ---)
      #    awk: count --- markers; body is what comes after c==2.
      local body
      body=$(awk '/^---$/{c++; next} c>=2 {print}' "$state_file" 2>/dev/null || echo "")

      # 3. Compose new file: front matter + body + new audit line appended
      local tmp
      tmp=$(mktemp -t godmode-state.XXXXXX) || return 1
      {
        printf '%s\n' "$new_fm"
        printf '\n'
        if [ -n "$body" ]; then
          printf '%s\n' "$body"
        else
          printf '# Audit Log\n\n'
        fi
        printf -- '- %s — %s\n' "$today" "$audit_line"
      } > "$tmp"

      # 4. Atomic replace (POSIX rename = atomic on same filesystem)
      mv "$tmp" "$state_file"
    }
    ```

    Notes:
    - `set -euo pipefail` is fine here because state.sh is sourced ONCE then the function executes in caller context. Use `local` everywhere.
    - The awk extracts EVERYTHING after the second `---`; this preserves the user's hand-edited audit-log body verbatim (D-16 append-only contract).
    - `jq -nr` (raw output) is required because we're constructing a multi-line string literal in jq.
    - The interpolated YAML uses `\"` for quoting `last_activity` (which may contain spaces, em dashes, etc.).
    - DO NOT trim/normalize prior audit lines — D-16 says skills NEVER edit prior audit lines.
    - `mv` is atomic on same filesystem (POSIX rename(2) semantics). Caveat documented in RESEARCH § Assumptions Log A2 (NFS / Windows-mounted filesystems are not atomic — WSL2 + native macOS/Linux are POSIX).
