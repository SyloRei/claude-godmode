# tests/fixtures/hooks/setup-fixtures.sh
# Creates 5 temp git repos with adversarial branch names; emits cwd JSON fixtures.
# Output: prints fixture paths (one per line) on success.
set -euo pipefail

FIXTURE_BASE="${TMPDIR:-/tmp}/godmode-hook-fixtures-$$"
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
mkdir -p "$FIXTURE_BASE"

create_repo() {
  local name="$1" branch="$2"
  local repo="$FIXTURE_BASE/$name"
  mkdir -p "$repo"
  (cd "$repo" && git init -q && git checkout -q -b "$branch" 2>/dev/null && \
   git -c user.email=test@example.com -c user.name=test commit --allow-empty -q -m init)
  printf '{"cwd":"%s"}\n' "$repo" > "$SCRIPT_DIR/cwd-${name}.json"

}
